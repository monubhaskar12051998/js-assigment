1) Create 2 objects - parent and child. Attach methods to parent and use those methods in child object using parents prototype

        const Parent = {
            details(){
                console.log(`Hey this is ${this.name}! Check out my latest single ${this.song} from my 10th studio album ${this.album}`);
            }
        }
        
        const child = Object.create(Parent);
        child.name = "Taylor Swift";
        child.song = "Lavender Haze";
        child.album = "Midnights";
        child.details();

        
        OUTPUT-Hey this is Taylor Swift! Check out my latest single Lavender Haze from my 10th studio album Midnights


2) Write code to explain prototype chaining 

        const objMusic = {
            artist: "Taylor Swift",
            genre: function(){
                console.log("Belongs to Indie genre");
            }
        }
        const album1 = Object.create(objMusic);
        album1.name = "Folklore";
        album1.year = "2020";
        console.log(album1.name);
        console.log(album1.year);
        console.log(album1.artist);     
        album1.genre();                 
        console.log("\n");
        const album2 = Object.create(album1);
        album2.name = "Evermore";
        console.log(album2.name);
        console.log(album2.year);      
        console.log(album2.artist);    
        album2.genre();

3) Add a method to calculate sum of all elements in Array in array's prototype, use that method to calculate sum for multiple arrays 

        let newArr = new Array(9,37,269);
        newArr.__proto__.add = function(arr){
            let count=0;
            for(let i=0; i< arr.length; i++)
            {
                count += arr[i];
            }
            return count;
        }

        let ans = newArr.add(newArr);
        console.log(ans);
        
        
        OUTPUT-315
